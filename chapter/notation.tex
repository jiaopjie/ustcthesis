\begin{notation}

\noindent\textsf{对数}\\
\begin{tabular}{@{}p{8em}ll}
  $\log_yx$ & 以 $y$ 为底 $x$ 的对数 \\
  $\ln x$   & 自然对数 $\log_ex$ \\
  $\lg x$   & 常用对数 $\log_{10}x$ \\
\end{tabular}

\vskip2em
\noindent\textsf{同余式}\\
\begin{tabular}{@{}p{8em}ll}
  $x\mod y$ & $x$ 的模 $y$ 剩余类 \\
\end{tabular}

\end{notation}
